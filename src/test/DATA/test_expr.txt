   i,j:INTEGER;
   k:FLOAT;

PROCEDURE a(q:INTEGER):INTEGER;

END a ;

PROCEDURE fact(n:INTEGER):INTEGER;
BEGIN
  IF n=0 THEN RETURN 1 END;
  IF n=1 THEN
      RETURN 1
  ELSE
      RETURN fact(n-1)*n
  END;
END fact ;
